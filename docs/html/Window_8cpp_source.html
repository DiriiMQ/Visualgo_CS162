60void Window::pollEvent() {
61 // event polling
62 while (this->window->pollEvent(this->event)) {
63 switch (this->event.type) {
64 case sf::Event::Closed:
65 this->window->close();
66 break;
67 case sf::Event::KeyPressed:
68 if (this->event.key.code == sf::Keyboard::Q) {
69 std::cout << "You have pressed Q!\n";
70 }
71 if (this->event.key.code == sf::Keyboard::W) {
72 std::cout << "You have pressed W!\n";
73 }
74 break;
75 default:
76 break;
77 }
78
79 if (this->submenuButton->pollEvent(this->mousePosView)) {
80 std::cout << "You have clicked on submenu button!\n";
81 this->scenes[this->currentScene]->isMenuOpen = (this->submenuButton->getTextString() == "<");
82 }
83
84 if (this->demoCodeButton->pollEvent(this->mousePosView)) {
85 std::cout << "You have clicked on demo code button!\n";
86 this->scenes[this->currentScene]->isDemoCodeOpen = (this->demoCodeButton->getTextString() == ">");
87 }
88
89 for (int i = 1; i < constants::sceneVariables::SCENE_COUNT; i++) {
90 if (this->scenes[i]->modeButton->pollEvent(this->mousePosView)) {
91 std::cout << "You have clicked on " << constants::sceneVariables::SCENE_NAMES[i] << " scene!\n";
92 this->currentScene = static_cast<constants::sceneVariables::Scene>(i);
93 this->scenes[this->currentScene]->isMenuOpen = (this->submenuButton->getTextString() == "<");
94 this->scenes[this->currentScene]->isDemoCodeOpen = (this->demoCodeButton->getTextString() == ">");
95 }
96 }
97
98 this->scenes[this->currentScene]->pollEvent(this->event, this->mousePosView);
99 }
100}
