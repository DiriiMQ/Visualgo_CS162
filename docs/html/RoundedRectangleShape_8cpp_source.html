79 sf::Vector2f RoundedRectangleShape::getPoint(std::size_t index) const
80 {
81 if(index >= myCornerPointCount*4)
82 return sf::Vector2f(0,0);
83
84 float deltaAngle = 90.0f/(myCornerPointCount-1);
85 sf::Vector2f center;
86 unsigned int centerIndex = index/myCornerPointCount;
87 static const float pi = 3.141592654f;
88
89 switch(centerIndex)
90 {
91 case 0: center.x = mySize.x - myRadius; center.y = myRadius; break;
92 case 1: center.x = myRadius; center.y = myRadius; break;
93 case 2: center.x = myRadius; center.y = mySize.y - myRadius; break;
94 case 3: center.x = mySize.x - myRadius; center.y = mySize.y - myRadius; break;
95 }
96
97 return sf::Vector2f(myRadius*cos(deltaAngle*(index-centerIndex)*pi/180)+center.x,
98 -myRadius*sin(deltaAngle*(index-centerIndex)*pi/180)+center.y);
99 }
