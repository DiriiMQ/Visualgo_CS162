11void StackScene::update() {
12 if (this->isMenuOpen) {
13 this->menu->update();
14
15 constants::MenuDataStructure::Button status = this->menu->getActiveOptionMenu();
16 constants::MenuDataStructure::CreateMode::Button createMode;
17 switch (status) {
18 case constants::MenuDataStructure::Button::CREATE_BUTTON:
19 createMode = this->menu->getActiveCreateMode();
20 if (createMode == constants::MenuDataStructure::CreateMode::Button::RANDOM_BUTTON) {
21 if (this->menu->createModeValue[0] == "None")
22 break;
23 if (this->menu->createModeValue[0].empty())
24 this->menu->createModeValue[0] = "0";
25 int size = std::stoi(this->menu->createModeValue[0]);
26 this->linkedList->createLinkedList(size);
27 } else if (createMode == constants::MenuDataStructure::CreateMode::Button::DEFINED_LIST_BUTTON) {
28 if (this->menu->createModeValue[1] == "None")
29 break;
30 std::vector<std::string> values;
31 std::string value = this->menu->createModeValue[1];
32 std::stringstream ss(value);
33 std::string token;
34 while (std::getline(ss, token, ',')) {
35 values.push_back(token);
36 }
37 this->linkedList->createLinkedList(values);
38 } else if (createMode == constants::MenuDataStructure::CreateMode::Button::FILE_BUTTON) {
39 if (this->menu->createModeValue[2] == "None")
40 break;
41 std::vector<std::string> values;
42 std::string value = this->menu->createModeValue[2];
43 std::stringstream ss(value);
44 std::string token;
45 while (std::getline(ss, token, ','))
46 values.push_back(token);
47 this->linkedList->createLinkedList(values);
48 this->menu->createModeValue[2] = "None";
49 }
50 this->controlMenu->reset();
51 break;
52 case constants::MenuDataStructure::Button::PUSH_BUTTON:
53 if (this->menu->pushModeValue == "None")
54 break;
55
56 this->linkedList->addNode(
57 0,
58 this->menu->pushModeValue,
59 this->pushModeEvents(0)
60 );
61
62 std::cout << "Pushed " << this->menu->pushModeValue << std::endl;
63 this->menu->pushModeValue = "None";
64 this->controlMenu->reset();
65 break;
66 case constants::MenuDataStructure::Button::POP_BUTTON:
67 if (this->menu->getActiveOptionMenu() != constants::MenuDataStructure::Button::POP_BUTTON)
68 break;
69
70 this->linkedList->deleteNode(
71 0,
72 this->popModeEvents(0)
73 );
74
75 std::cout << "Popped " << std::endl;
76 this->menu->resetActiveOptionMenuOnly();
77 this->controlMenu->reset();
78 break;
79 case constants::MenuDataStructure::Button::CLEAR_BUTTON:
80 if (this->menu->getActiveOptionMenu() != constants::MenuDataStructure::Button::CLEAR_BUTTON)
81 break;
82
83 this->linkedList->createLinkedList(0);
84
85 std::cout << "Cleared " << std::endl;
86 this->menu->resetActiveOptionMenuOnly();
87 this->controlMenu->reset();
88 break;
89 }
90 }
91
92 this->controlMenu->update();
93
94 this->linkedList->processControlMenu(this->controlMenu->getStatus());
95 this->linkedList->setSpeed(this->controlMenu->getSpeed());
96
97 this->linkedList->update();
98}
