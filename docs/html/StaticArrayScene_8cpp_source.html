11void StaticArrayScene::update() {

15 constants::MenuArray::Button status = this->menu->getActiveOptionMenu();
16 constants::MenuArray::CreateMode::Button createMode;
17 switch (status){
18 case constants::MenuArray::Button::CREATE_BUTTON:
19 createMode = this->menu->getActiveCreateMode();
20 if (createMode == constants::MenuArray::CreateMode::Button::RANDOM_BUTTON) {
21 if (this->menu->createModeValue[0] == "None")
22 break;
23 if (this->menu->createModeValue[0].empty())
24 this->menu->createModeValue[0] = "0";
25 int size = std::stoi(this->menu->createModeValue[0]);
26 this->array->createArray(size);
27 } else if (createMode == constants::MenuArray::CreateMode::Button::DEFINED_LIST_BUTTON) {
28 if (this->menu->createModeValue[1] == "None")
29 break;
30 std::vector<std::string> values;
31 std::string value = this->menu->createModeValue[1];
32 std::stringstream ss(value);
33 std::string token;
34 while (std::getline(ss, token, ',')) {
35 values.push_back(token);
36 }
37 this->array->createArray(values);
38 } else if (createMode == constants::MenuArray::CreateMode::Button::FILE_BUTTON) {
39 if (this->menu->createModeValue[2] == "None")
40 break;
41 std::vector<std::string> values;
42 std::string value = this->menu->createModeValue[2];
43 std::stringstream ss(value);
44 std::string token;
45 while (std::getline(ss, token, ','))
46 values.push_back(token);
47 this->array->createArray(values);
48 this->menu->createModeValue[2] = "None";
49 }
50 this->controlMenu->reset();
51 break;
52 case constants::MenuArray::Button::ADD_BUTTON:
53 if (this->menu->addModeValue[0] == "None" || this->menu->addModeValue[1] == "None" || this->menu->addModeValue[0].empty())
54 break;
55
56 this->array->addSquare(
57 std::stoi(this->menu->addModeValue[0]),
58 this->menu->addModeValue[1],
59 this->addModeEvents(std::stoi(this->menu->addModeValue[0]))
60 );
61
62 std::cout << "Add: " << this->menu->addModeValue[0] << " " << this->menu->addModeValue[1] << std::endl;
63 this->menu->addModeValue[0] = this->menu->addModeValue[1] = "None";
64 this->controlMenu->reset();
65 break;
66 case constants::MenuArray::Button::DELETE_BUTTON:
67 if (this->menu->deleteModeValue == "None" || this->menu->deleteModeValue.empty())
68 break;
69
70 this->array->deleteSquare(
71 std::stoi(this->menu->deleteModeValue),

77 this->controlMenu->reset();
78 break;
79 case constants::MenuArray::Button::UPDATE_BUTTON:
