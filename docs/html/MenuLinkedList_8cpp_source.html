44void MenuLinkedList::pollEvents(sf::Event event, sf::Vector2f mousePosView) {
45 if (this->activeOptionMenu != constants::MenuLinkedList::Button::NONE)
46 this->buttons[this->activeOptionMenu]->setColor(constants::normalGray);
47
48 for (int i = 0; i < constants::MenuLinkedList::BUTTON_COUNT; i++) {

193void MenuLinkedList::updateCreateMode() {
194 if (this->activeCreateMode != constants::MenuLinkedList::CreateMode::Button::NONE)
195 this->subCreateMode[this->activeCreateMode]->setColor(constants::clickGreen);
196
197 for (Button* button : this->subCreateMode) {
198 button->update();
199 }
200
201// this->testTextbox->update();
202 if (this->activeCreateMode < constants::MenuLinkedList::CreateMode::TEXTBOX_COUNT) {
203 this->createTextbox[this->activeCreateMode]->update();
204 std::string inputUser = this->createTextbox[this->activeCreateMode]->getTextString();
205 if (inputUser != "None") {
206 std::cout << inputUser << std::endl;
207 this->createTextbox[this->activeCreateMode]->resetInput();
208 }
209 this->createModeValue[this->activeCreateMode] = inputUser;
210 } else if (this->activeCreateMode == constants::MenuLinkedList::CreateMode::FILE_BUTTON) {
211 if (this->isOpenFileDialog) {
212 auto f = pfd::open_file("Choose files to read", pfd::path::home(),
213 {"Text Files (.txt .text)", "*.txt *.text",
214 "All Files", "*"});
215
216 // wait for the user to select a file unless the window will be not responsive
217 while (!f.ready(100)) {
218 sf::Event event{};
219 this->window->pollEvent(event);
220 }
221
222 if (!f.result().empty()) {
223 std::ifstream file(f.result()[0]);
224 std::string line;
225 file >> line;
226 this->createModeValue[this->activeCreateMode] = line;
227 }
228 }
229 this->isOpenFileDialog = false;
230 }
