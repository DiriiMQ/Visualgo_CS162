67void Array::processControlMenu(ControlMenu::StatusCode status) {
68 if (this->clock.getElapsedTime().asSeconds() < this->delayTime / this->speed)
69 return;
70 switch (status){
71 case ControlMenu::StatusCode::PREVIOUS:
72 if (this->currentEvent > 0)
73 --this->currentEvent;
74 break;
75 case ControlMenu::StatusCode::PAUSE:
76// std::cout << "PAUSE" << std::endl;
77 break;
78 case ControlMenu::StatusCode::PLAY:
79 if (this->currentEvent + 1 < this->events.size()) {
80 this->isDelay = true;
81 this->clock.restart();
82 }
83 case ControlMenu::StatusCode::NEXT:
84 if (this->currentEvent + 1 < this->events.size())
85 ++this->currentEvent;
86 break;
87 default:
88 break;
89 }
90}

105void Array::updateAnimation() {
106 if (this->squares.empty())
107 return;
108
109 for (auto &square : this->squares) {
110 square->reset();
111 }
112 for (auto &square : this->squaresTemp) {
113 square->reset();
114 }
115
116 EventAnimation &event = this->events[this->currentEvent];
117 for (int i = 0; i < event.eventSquares.size(); ++i) {
118 this->squares[i]->setStatus(event.eventSquares[i].status);
119 this->squares[i]->setPrintPreVal(event.eventSquares[i].isPrintPreVal);
120 this->squares[i]->setTitle(event.eventSquares[i].title);
121 }
122 for (int i = 0; i < event.eventSquaresTemp.size(); ++i) {
123 this->squaresTemp[i]->setStatus(event.eventSquaresTemp[i].status);
124 this->squaresTemp[i]->setPrintPreVal(event.eventSquaresTemp[i].isPrintPreVal);
125 this->squaresTemp[i]->setTitle(event.eventSquaresTemp[i].title);
126 }
127
128 if (this->highlighter)
129 this->highlighter->toggle(event.lines);
130
131 for (auto &square : this->squares) {
132 square->update();
133 }
134 for (auto &square : this->squaresTemp) {
135 square->update();
136 }

139void Array::resetEvents() {

144 this->squaresTemp.clear();

147 this->squares.pop_back();
148
149 for (int i = 0; i < this->size; ++i)
150 this->squares[i]->setStatus(Square::Status::active);
151 for (int i = this->size; i < this->squares.size(); ++i)
152 this->squares[i]->setStatus(Square::Status::inactive);
153 if (this->size)
154 this->squares[this->size - 1]->setTitle("n");
155}
