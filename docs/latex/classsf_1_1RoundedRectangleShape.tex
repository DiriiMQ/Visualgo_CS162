\doxysubsection{Detailed Description}
Specialized shape representing a rectangle with rounded corners. 

This class inherits all the functions of sf\+::\+Transformable (position, rotation, scale, bounds, ...) as well as the functions of sf\+::\+Shape (outline, color, texture, ...).

Usage example\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\mbox{\hyperlink{classsf_1_1RoundedRectangleShape}{sf::RoundedRectangleShape}}\ roundedRectangle;}
\DoxyCodeLine{rectangle.\mbox{\hyperlink{classsf_1_1RoundedRectangleShape_a6c768c97f3b1412f8b310c3c64b7cc91}{setSize}}(sf::Vector2f(100,\ 50));}
\DoxyCodeLine{rectangle.setCornersRadius(5);}
\DoxyCodeLine{rectangle.setOutlineThickness(5);}
\DoxyCodeLine{rectangle.setPosition(10,\ 20);}
\DoxyCodeLine{...}
\DoxyCodeLine{window.draw(rectangle);}

\end{DoxyCode}
